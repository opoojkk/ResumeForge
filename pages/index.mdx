import Section from '../components/Section'
import Timeline from '../components/Timeline'
import SkillBar from '../components/SkillBar'
import Tags from '../components/Tags'
import Header from '../components/Header'
import ResumeEditor from '../components/ResumeEditor'
import ResizableLayout from '../components/ResizableLayout'
import resumeData from '../data/resume-data.json'
import { getContactIcon } from '../lib/iconHelper'
import { useState, useEffect, useCallback, useRef } from 'react'

export const meta = {
  title: '刘建东 - 个人简历',
  description: 'Android 开发工程师'
}

export function ResumeContent() {
  const STORAGE_KEY = 'resume-forge-data'
  const AVATAR_CACHE_KEY = 'resume-forge-avatar'
  const JSON_SYNC_DELAY = 500
  const normalizeResumeData = (jsonData) => {
    const nextData = JSON.parse(JSON.stringify(jsonData ?? {}))

    if (!nextData.personalInfo) {
      nextData.personalInfo = {}
    }

    if (!nextData.exportSettings || typeof nextData.exportSettings !== 'object') {
      nextData.exportSettings = {}
    }

    if (typeof nextData.exportSettings.pdfFileName !== 'string') {
      nextData.exportSettings.pdfFileName = ''
    }

    return nextData
  }

  const migrateContactsData = (jsonData) => {
    if (Array.isArray(jsonData.personalInfo?.contacts)) {
      const contactsArray = jsonData.personalInfo.contacts
      const contactsObj = {
        email: '',
        phone: '',
        github: { title: '', link: '' },
        blog: { title: '', link: '' },
      }

      contactsArray.forEach((contact) => {
        if (contact.type === 'email') {
          contactsObj.email = contact.text
        } else if (contact.type === 'phone') {
          contactsObj.phone = contact.text
        } else if (contact.type === 'github') {
          contactsObj.github = {
            title: contact.title || '',
            link: contact.link || `https://github.com/${contact.text}`,
          }
        } else if (contact.type === 'website') {
          contactsObj.blog = {
            title: contact.title || '',
            link: contact.link || contact.text,
          }
        }
      })

      jsonData.personalInfo.contacts = contactsObj
    } else if (jsonData.personalInfo?.contacts) {
      const contacts = jsonData.personalInfo.contacts

      if (typeof contacts.github === 'string') {
        contacts.github = {
          title: '',
          link: contacts.github,
        }
      } else if (!contacts.github) {
        contacts.github = { title: '', link: '' }
      }

      if (typeof contacts.blog === 'string') {
        contacts.blog = {
          title: '',
          link: contacts.blog,
        }
      } else if (!contacts.blog) {
        contacts.blog = { title: '', link: '' }
      }
    }

    return jsonData
  }

  const sanitizePdfFileName = (value) => (
    (value || '')
      .replace(/[<>:"/\\|?*\u0000-\u001F]/g, '')
      .replace(/\.pdf$/i, '')
      .replace(/\s+/g, ' ')
      .trim()
  )

  const getPdfFileName = (resumeData) => {
    const customName = sanitizePdfFileName(resumeData?.exportSettings?.pdfFileName)
    if (customName) {
      return customName
    }

    const fallbackName = sanitizePdfFileName(
      resumeData?.personalInfo?.name
        ? `${resumeData.personalInfo.name}-简历`
        : 'resume'
    )

    return fallbackName || 'resume'
  }

  const [data, setData] = useState(() => normalizeResumeData(resumeData))
  const [selectedJsonFileName, setSelectedJsonFileName] = useState('')
  const [jsonSyncStatus, setJsonSyncStatus] = useState('idle')
  const resumePrintRef = useRef(null)
  const jsonFileHandleRef = useRef(null)
  const jsonSyncTimeoutRef = useRef(null)

  const persistToLocalCache = useCallback((normalizedData) => {
    try {
      const dataToSave = JSON.parse(JSON.stringify(normalizedData))
      if (dataToSave.personalInfo?.avatar?.startsWith('data:')) {
        localStorage.setItem(AVATAR_CACHE_KEY, dataToSave.personalInfo.avatar)
        dataToSave.personalInfo.avatar = '__LOCAL_AVATAR__'
      } else {
        localStorage.removeItem(AVATAR_CACHE_KEY)
      }
      localStorage.setItem(STORAGE_KEY, JSON.stringify(dataToSave))
    } catch (e) {
      // ignore
    }
  }, [])

  const syncJsonFile = useCallback(async (nextData) => {
    const handle = jsonFileHandleRef.current
    if (!handle) return

    setJsonSyncStatus('syncing')

    try {
      if (typeof handle.requestPermission === 'function') {
        const permission = await handle.requestPermission({ mode: 'readwrite' })
        if (permission !== 'granted') {
          throw new Error('write permission denied')
        }
      }

      const writable = await handle.createWritable()
      await writable.write(JSON.stringify(nextData, null, 2))
      await writable.close()
      setJsonSyncStatus('synced')
    } catch (error) {
      setJsonSyncStatus('error')
    }
  }, [])

  const scheduleJsonFileSync = useCallback((nextData, immediate = false) => {
    if (!jsonFileHandleRef.current) return

    if (jsonSyncTimeoutRef.current) {
      window.clearTimeout(jsonSyncTimeoutRef.current)
      jsonSyncTimeoutRef.current = null
    }

    if (immediate) {
      void syncJsonFile(nextData)
      return
    }

    setJsonSyncStatus('syncing')
    jsonSyncTimeoutRef.current = window.setTimeout(() => {
      jsonSyncTimeoutRef.current = null
      void syncJsonFile(nextData)
    }, JSON_SYNC_DELAY)
  }, [syncJsonFile])

  // 客户端挂载后从 localStorage 恢复数据
  useEffect(() => {
    try {
      const cached = localStorage.getItem(STORAGE_KEY)
      if (cached) {
        const parsed = normalizeResumeData(JSON.parse(cached))
        // 恢复单独缓存的头像（base64 数据较大，单独存储）
        const cachedAvatar = localStorage.getItem(AVATAR_CACHE_KEY)
        if (cachedAvatar && parsed.personalInfo) {
          parsed.personalInfo.avatar = cachedAvatar
        }
        setData(parsed)
      }
    } catch (e) {
      // ignore
    }
  }, [])

  useEffect(() => {
    return () => {
      if (jsonSyncTimeoutRef.current) {
        window.clearTimeout(jsonSyncTimeoutRef.current)
      }
    }
  }, [])

  const handleDataChange = useCallback((newData) => {
    const normalizedData = normalizeResumeData(newData)
    setData(normalizedData)
    persistToLocalCache(normalizedData)
    scheduleJsonFileSync(normalizedData)
  }, [persistToLocalCache, scheduleJsonFileSync])

  const handleReset = useCallback(() => {
    localStorage.removeItem(STORAGE_KEY)
    localStorage.removeItem(AVATAR_CACHE_KEY)
    const defaultData = normalizeResumeData(resumeData)
    setData(defaultData)
    scheduleJsonFileSync(defaultData, true)
  }, [scheduleJsonFileSync])

  const handleSelectJsonFile = useCallback(async () => {
    if (typeof window === 'undefined' || typeof window.showOpenFilePicker !== 'function') {
      alert('当前浏览器不支持直接选择并同步本地 JSON 文件，请使用基于 Chromium 的浏览器。')
      return
    }

    try {
      const [fileHandle] = await window.showOpenFilePicker({
        multiple: false,
        excludeAcceptAllOption: true,
        types: [
          {
            description: 'JSON 文件',
            accept: {
              'application/json': ['.json'],
            },
          },
        ],
      })

      if (!fileHandle) return

      if (typeof fileHandle.requestPermission === 'function') {
        const permission = await fileHandle.requestPermission({ mode: 'readwrite' })
        if (permission !== 'granted') {
          alert('需要授予该文件的读写权限后，才能启用自动同步。')
          return
        }
      }

      const file = await fileHandle.getFile()
      const text = await file.text()
      const parsed = normalizeResumeData(migrateContactsData(JSON.parse(text)))

      jsonFileHandleRef.current = fileHandle
      setSelectedJsonFileName(file.name)
      setJsonSyncStatus('synced')
      setData(parsed)
      persistToLocalCache(parsed)
    } catch (error) {
      if (error?.name === 'AbortError') {
        return
      }

      setJsonSyncStatus('error')
      alert('读取 JSON 文件失败，请确认文件格式正确。')
    }
  }, [persistToLocalCache])

  const getPrintableStyles = useCallback(() => {
    return Array.from(document.styleSheets)
      .map((sheet) => {
        try {
          return Array.from(sheet.cssRules)
            .map((rule) => rule.cssText)
            .join('\n')
        } catch {
          return ''
        }
      })
      .join('\n')
  }, [])
  
  const handleExportPDF = useCallback(async () => {
    const resumeElement = resumePrintRef.current

    if (!resumeElement) {
      return
    }

    const printFrame = document.createElement('iframe')
    printFrame.style.position = 'fixed'
    printFrame.style.right = '0'
    printFrame.style.bottom = '0'
    printFrame.style.width = '0'
    printFrame.style.height = '0'
    printFrame.style.border = '0'
    printFrame.style.visibility = 'hidden'
    document.body.appendChild(printFrame)

    const frameWindow = printFrame.contentWindow
    if (!frameWindow) {
      printFrame.remove()
      return
    }

    const clonedResume = resumeElement.cloneNode(true)
    const documentTitle = getPdfFileName(data)

    frameWindow.document.open()
    frameWindow.document.write(`<!DOCTYPE html>
<html lang="zh-CN">
  <head>
    <meta charset="utf-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1" />
    <title>${documentTitle}</title>
    <style>
      @page {
        size: A4;
        margin: 0;
      }

      * {
        box-sizing: border-box;
      }

      html, body {
        margin: 0;
        padding: 0;
        width: 100%;
        background: #ffffff;
      }

      body {
        -webkit-print-color-adjust: exact;
        print-color-adjust: exact;
      }

      #print-root {
        width: 210mm;
        min-height: 297mm;
        margin: 0 auto;
        padding: 0;
        background: #ffffff;
      }

      #print-root .resume-container {
        width: 100% !important;
        max-width: 100% !important;
        margin: 0 !important;
        background: #ffffff !important;
        box-shadow: none !important;
        border-radius: 0 !important;
      }

      ${getPrintableStyles()}
    </style>
  </head>
  <body>
    <div id="print-root">${clonedResume.outerHTML}</div>
  </body>
</html>`)
    frameWindow.document.close()

    const waitForPrintReady = async () => {
      const frameDocument = frameWindow.document

      if (frameDocument.fonts?.ready) {
        await frameDocument.fonts.ready
      }

      const images = Array.from(frameDocument.images)
      await Promise.all(
        images
          .filter((img) => !img.complete)
          .map((img) => new Promise((resolve) => {
            img.addEventListener('load', () => resolve(), { once: true })
            img.addEventListener('error', () => resolve(), { once: true })
          }))
      )

      await new Promise((resolve) => {
        frameWindow.requestAnimationFrame(() => {
          frameWindow.requestAnimationFrame(resolve)
        })
      })
    }

    try {
      await waitForPrintReady()
      frameWindow.focus()
      frameWindow.print()
      frameWindow.onafterprint = () => {
        printFrame.remove()
      }
      window.setTimeout(() => {
        if (document.body.contains(printFrame)) {
          printFrame.remove()
        }
      }, 1500)
    } catch {
      printFrame.remove()
    }
  }, [data, getPrintableStyles])
  
  return (
    <ResizableLayout
      editor={
        <ResumeEditor
          initialData={data}
          onDataChange={handleDataChange}
          onReset={handleReset}
          onSelectJsonFile={handleSelectJsonFile}
          selectedJsonFileName={selectedJsonFileName}
          jsonSyncStatus={jsonSyncStatus}
        />
      }
      preview={
        <div className="preview-panel">
          <div className="preview-toolbar">
            <div className="preview-toolbar-copy">
              <p className="preview-kicker">Live Preview</p>
              <h3>简历预览</h3>
              <span>导出范围固定为右侧简历预览区域</span>
            </div>
            <div className="preview-toolbar-actions">
              <button className="btn-export-pdf-floating" onClick={handleExportPDF}>
                📄 导出 PDF
              </button>
            </div>
          </div>
          <div className="resume-preview">
            <div className="resume-container" ref={resumePrintRef}>
              <Header 
                name={data.personalInfo.name}
                title={data.personalInfo.title}
                avatar={data.personalInfo.avatar}
                location={data.personalInfo.location}
                contacts={[
                  data.personalInfo.contacts.blog?.link && {
                    icon: getContactIcon('website'),
                    text: data.personalInfo.contacts.blog.title || 
                          data.personalInfo.contacts.blog.link.replace(/^https?:\/\//, ''),
                    link: data.personalInfo.contacts.blog.link
                  },
                  data.personalInfo.contacts.phone && {
                    icon: getContactIcon('phone'),
                    text: data.personalInfo.contacts.phone
                  },
                  data.personalInfo.contacts.github?.link && {
                    icon: getContactIcon('github'),
                    text: data.personalInfo.contacts.github.title || 
                          data.personalInfo.contacts.github.link.replace(/^https?:\/\//, ''),
                    link: data.personalInfo.contacts.github.link
                  },
                  data.personalInfo.contacts.email && {
                    icon: getContactIcon('email'),
                    text: data.personalInfo.contacts.email,
                    link: `mailto:${data.personalInfo.contacts.email}`
                  }
                ].filter(Boolean)}
              />

              <Section title="教育经历">
                {data.education.map((edu, index) => (
                  <div key={index} className="education-item">
                    <div className="education-header">
                      <div className="education-info">
                        <h3>{edu.school} · {edu.major}</h3>
                      </div>
                      <span className="education-time">{edu.startDate} - {edu.endDate}</span>
                    </div>
                    <p className="education-school">{edu.degree}</p>
                  </div>
                ))}
              </Section>

              <Section title="工作经历">
                <div className="work-experience">
                  {data.workExperience.map((work, index) => (
                    <div key={index} className="company-block">
                      <div className="company-header">
                        <div className="company-info">
                          <h3 className="company-name">{work.company}</h3>
                          <span className="company-location">{work.location}</span>
                        </div>
                        <span className="work-time">{work.startDate} - {work.endDate}</span>
                      </div>
                      <p className="job-title">{work.position}</p>
                    
                      {work.projects.map((project, projectIndex) => (
                        <div key={projectIndex} className="project-block">
                          <h4 className="project-name">{project.name}</h4>
                          {project.description && <p className="project-desc">{project.description}</p>}
                          <ul className="project-details">
                            {project.responsibilities.map((resp, respIndex) => (
                              <li key={respIndex}>{resp}</li>
                            ))}
                          </ul>
                        </div>
                      ))}
                    </div>
                  ))}
                </div>
              </Section>

              <Section title="专业技能">
                <div className="skills-section">
                  {data.skills.map((skill, index) => (
                    <div key={index} className="skill-group">
                      <h4 className="skill-group-title">{skill}</h4>
                    </div>
                  ))}
                </div>
              </Section>

              <Section title="开源项目">
                <div className="opensource-projects">
                  {data.openSourceProjects.map((project, index) => (
                    <div key={index} className="project-item">
                      <h3 className="project-title">
                        <a href={project.link} target="_blank" rel="noopener noreferrer">{project.name}</a>
                      </h3>
                      <p className="project-desc">{project.description}</p>
                    </div>
                  ))}
                </div>
              </Section>

              <Section title="文章">
                <div className="articles-list">
                  {data.articles.map((article, index) => (
                    <div key={index} className="article-item">
                      <h3 className="article-title">
                        <a href={article.link} target="_blank" rel="noopener noreferrer">{article.title}</a>
                      </h3>
                    </div>
                  ))}
                </div>
              </Section>
            </div>
          </div>
        </div>
      }
    />
  )
}

<ResumeContent />
